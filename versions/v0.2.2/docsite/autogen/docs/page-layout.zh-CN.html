页面布局

因与 vue-router 结合后, 文档中无法展示 Demo

伪代码如下, LayoutComponent 组件代码参考下方布局示例

// main.js
import xlucky from 'xlucky.vue2';
Vue.use(xlucky);

const routes = [
  { path: '/login', component: LoginComponent }
  {
    path: '/account-manage', component: LayoutComponent,
    children: [
      { path: '/account-manage/account', component: AccountViewComponent },
      { path: '/account-manage/role', component: RoleViewComponent },
      { path: '/account-manage/right', component: RightViewComponent },
      {
        path: '/account-manage/other-module',
        component: Vue.component('RouterView'),
        children: [ /* 子页面集合 */ ]
      },
    ]
  }
];

new Vue({
  router: new VueRouter({ routes }),
  store,
  render: (h) => h(App),
}).$mount('#app');
